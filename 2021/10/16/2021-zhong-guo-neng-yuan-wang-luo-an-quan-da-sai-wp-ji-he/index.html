    if i[0] == 'b':
        i = i[1:]
        i = int(i,2)
        i = chr(i)
        result.append(i)
    elif i[0] == '0':
        i = i[1:]
        i = int(i,8)
        i = chr(i)
        result.append(i)
    elif i[0] == 'x':
        i = int("0"+i, 16)
        i = chr(i)
        result.append(i)
    else:
        i = base64.b64decode(i)
        i = bytes.decode(i)
        result.append(i)


flag = ''.join(result)
print(flag)
